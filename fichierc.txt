#include "fichier.h"



int chargerNiveau(int niveau[][NB_BLOC_HAUTEUR])
{
FILE* fichier = NULL;
char ligneFichier[NB_BLOC_LARGEUR * NB_BLOC_HAUTEUR + 1] = {0};
int i = 0, j = 0;

fichier = fopen("niveaux.lvl", "r");
if (fichier == NULL)
    return 0;

fgets(ligneFichier, NB_BLOC_LARGEUR * NB_BLOC_HAUTEUR + 1, fichier);// recupere la position des sprites dans ligneFichier
/*lit le fichier niveaux */ 
for (i = 0 ; i < NB_BLOC_HAUTEUR; i++)
{
    for (j = 0 ; j < NB_BLOC_LARGEUR; j++)
    {
        switch (ligneFichier[(i * NB_BLOC_LARGEUR) + j])
        {
            case '0':
                niveau[i][j] = 0;
                break;
            case '1':
                niveau[i][j] = 1;
                break;
            case '2':
                niveau[i][j] = 2;
                break;
            case '3':
                niveau[i][j] = 3;
                break;
            case '4':
                niveau[i][j] = 4;
                break;
        }
    }
}
fclose(fichier);
return 1;
}
